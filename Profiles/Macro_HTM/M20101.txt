// macroloop ptr citire senzori sloturi

// citim starea ledurilor de input ptr senzorii sloturilor
// folosim inputuri fixe, Port 5 si Pini 2-11
// preluam starea lor din LED-urile de DIAG (121-130)
for (int i = 0; i < 10; i++) {
		if (AS3.Getbuttonstate(20202)) { // daca senzorii sunt activati
			if (AS3.GetLED(i+801)) {
				AS3.SetLED((!AS3.GetLED(i+121)), (i+831));
			}
		} else {
			AS3.SetLED(false, (i+831));
		}				
	}
